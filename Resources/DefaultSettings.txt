; This is the default configuration for Noxico. Feel free to adjust this as you see fit.

[font]
;    family: The font family name, like "Consolas", "DejaVu Sans Mono", or "Fixedsys".
;            If the font can't be loaded, the system's default monospace font is used.
;      size: The size in points.
;      bold: Obvious.
;  x-adjust: Some fonts may need to be nudged. This and y-adjust will allow that.
;            Don't use too large values or things'll go wonky.
; cellwidth: This and cellheight can be used to override the automatic cell size
;            detection. If set to zero, the cell size will be (badly) determined
;            automatically.
; cleartype: Enables or disables font smoothing.
family=Consolas
size=11
bold=true
x-adjust=-2
y-adjust=-1
cellwidth=0
cellheight=0
cleartype=true;

[audio]
;     enabled: Should be pretty obvious. Volume controls are ignored if false.
; musicvolume: From 0 to 100, how loud the music should be.
; soundvolume: From 0 to 100, how loud sound effects should be.
enabled=true
musicvolume=100
soundvolume=100

[misc]
; rememberpause: Remember the active page on the Pause screen. If set to false,
;                opening the Pause screen will reset to the "Character stats"
;                page every time.
;      autorest: Specifies the timeout at which the game will automatically
;                press the Rest key, to give the illusion of realtime gameplay.
;                The lower, the faster.
;     skiptitle: Skip the title screen if there's already a game to restore.
;                Has no effect on a clean slate.
;    vistasaves: Save to %USERPROFILE%\Saved Games on Windows Vista and 7.
;                Has no effect on any other operating system.
;      savepath: Specifies where to save world data. Can be an absolute or
;                relative path. If preceded with "$", that part of the path
;                is taken to be %APPDATA% on Windows, or ~/.config on Linux.
;      shotpath: Specifies where to put screenshots. Same deal as savepath.
;     candytron: If your copy is built to support it, enables Candytron Mode,
;                a modern-looking fullscreen mode.
;       pngmode: If your copy is built to support it, enables PNG Font Mode,
;                which replaces the TTF renderer with a bitmap engine like the
;                first private releases had. If Candytron Mode is enabled,
;                PNG Mode is automatically disabled.
;       pngfont: Specifies which set of font bitmaps to use in PNG Mode.
rememberpause=true
autorest=100
skiptitle=false
vistasaves=true
savepath=$/Noxico/Saves
shotpath=$/Noxico/Screenshots
candytron=false
pngmode=true
pngfont=fixedsex

[filters]
; color: Specifies the special filter to apply to a cell's color information.
;        Valid options are:
;          none: Don't do anything - show the colors as they are defined.
;           cga: Convert the colors to the standard 16 color CGA palette. Sucks.
;          mono: Convert the colors to a 256-level grayscale.
;           psp: Convert the colors to match a Paintshop Pro .PspPalette file.
;  char: Specifies the special filter to apply to a cell's character information.
;        Valid options are:
;          none: Show all codepoints as specified.
;          437:  Convert all codepoints above 127 to their OEM US counterpart.
;		   7bit: Remove all codepoints above 127, use +-| for box drawing.
;   pal: Specifies which .PspPalette file to use. If not found, switches back to
;        color=none. Both sixteen and 256-color palettes are supported.
color=none
char=none
pal=Noxico.PspPalette
